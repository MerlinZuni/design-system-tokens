**Step 4: Implement utility functions:**

`tokenPathToCssVar(path: string, prefix: string = 'dsx'): string` — converts dot path to CSS var name:

`resolveAliasChain(alias: string, primitiveData: Record<string, unknown>): string[]` — given an alias like `{color.neutral.0}`, returns `['color.background.default', 'color.neutral.0', '#ffffff']`. `flattenTokens(obj: Record<string, unknown>, parentPath: string = ''): TokenRow[]` — recursively walks a DTCG JSON object, collecting leaf nodes (objects with `$value`). Skips `$type` and `$description` at group level. For each leaf: name = full dot path, rawValue = `$value`, description = `$description`.

**1. CopyToClipboard.tsx**

A wrapper component that copies text to clipboard on click with a 300ms background flash.

```typescript
import React, { useState, useCallback } from 'react'

interface CopyToClipboardProps {
  text: string
  children: React.ReactNode
}

export function CopyToClipboard({ text, children }: CopyToClipboardProps) {
  const [copied, setCopied] = useState(false)

  const handleClick = useCallback(async () => {
    await navigator.clipboard.writeText(text)
    setCopied(true)
    setTimeout(() => setCopied(false), 300)
  }, [text])

  return (
    <span
      onClick={handleClick}
      role="button"
      tabIndex={0}
      onKeyDown={(e) => e.key === 'Enter' && handleClick()}
      aria-label={`Copy ${text}`}
      style={{
        cursor: 'pointer',
        padding: '2px 4px',
        borderRadius: '3px',
        transition: 'background-color 0.15s',
        backgroundColor: copied ? 'var(--dsx-color-highlight-selected)' : 'transparent',
      }}
    >
      {children}
      {copied && <span className="sr-only" aria-live="polite">Copied to clipboard</span>}
    </span>
  )
}
```

**2. AliasChain.tsx**

Displays the full resolution path: `semantic.token -> primitive.token -> #rawvalue`.

```typescript
import React from 'react'

interface AliasChainProps {
  chain: string[]  // e.g. ['color.background.default', 'color.neutral.0', '#ffffff']
}

export function AliasChain({ chain }: AliasChainProps) {
  return (
    <span style={{
      display: 'inline-flex',
      alignItems: 'center',
      gap: '4px',
      fontSize: '12px',
      fontFamily: 'monospace',
      color: 'var(--dsx-color-text-muted)',
      backgroundColor: 'var(--dsx-color-surface-panel)',
      padding: '2px 6px',
      borderRadius: '4px',
    }}>
      {chain.map((segment, i) => (
        <React.Fragment key={i}>
          {i > 0 && <span style={{ color: 'var(--dsx-color-action-default)' }}>{' \u2192 '}</span>}
          <span>{segment}</span>
        </React.Fragment>
      ))}
    </span>
  )
}
```

The arrow glyph uses Unicode right arrow (U+2192). Arrow color is `var(--dsx-color-action-default)` (accent per UI spec).

**4. SemanticTokenSection.tsx**

A generic, reusable theme-reactive component that displays semantic tokens filtered by a prefix. This component will be reused by Spacing.mdx (Plan 03), Elevation.mdx (Plan 04), and can replace SemanticColorTable for simpler cases.

```typescript
import React from 'react'
import { useGlobals } from '@storybook/preview-api'
import { TokenTable } from './TokenTable'
import { getSemanticTokensForTheme } from './token-data'

interface SemanticTokenSectionProps {
  prefix: string  // e.g. "space.", "shadow.", "color."
}

export function SemanticTokenSection({ prefix }: SemanticTokenSectionProps) {
  const [globals] = useGlobals()
  const { brand = 'parent', mode = 'light' } = globals

  const tokens = getSemanticTokensForTheme(brand, mode)
    .filter((t) => t.name.startsWith(prefix))

  return <TokenTable tokens={tokens} showAliasChain={true} />
}
```
